function hasDuplicates(array) {
    var valuesSoFar = Object.create(null);
    for (var i = 0; i < array.length; ++i) {
        var value = array[i];
        if (value in valuesSoFar) {
            return true;
        }
        valuesSoFar[value] = true;
    }
    return false;
}

function verificaRepetido(array, codReporte) {

    var respuesta = false;
    for (i = 0; i < array.length; i++) {
        if (array[i] == codReporte) {
            respuesta = true;
        }
    }
    return respuesta;
}

function agregarBuscador(codRP) {
    $('#btnBusquedaTabla' + codRP).click(function () {
        $('#txtBuscar' + codRP).val('');

        if ($('#divInputBusqueda' + codRP).hasClass('active')) {
            $('#divInputBusqueda' + codRP).removeClass('active');
            $('#divInputBusqueda' + codRP).css('width', '39px');
            $.uiTableFilter($('.table'), '');
        } else {
            $('#divInputBusqueda' + codRP).addClass('active');
            $('#divInputBusqueda' + codRP).css('width', '239px');
            $('#txtBuscar' + codRP).focus().select();
        }
    })
    $('#txtBuscar' + codRP).keyup(function () {
        if (codRP == 'RP115') {
            $.uiTableFilter($(`#tb${codRP}`), this.value, ['PSJ']);
        } else {
            $.uiTableFilter($(`#tb${codRP}`), this.value);
        }
    });//tbRP115
}

function agregaOpcTodos(select) {
    var existeOpctodos = 0;

    select.find('option').each(function () {
        if (this.value == 0) {
            existeOpctodos = 1;
        }
    });

    if (existeOpctodos == 0) {
        select.append('<option value="0" selected>--- TODOS ---</option>');
    } else {
        select.val(0);
    }
}

}

function cerrarTab(numTab, spanClose, codReporte) {
    $('#trReporte' + codReporte + '').removeClass('hoverReporteTr');
    //removerArray(arrCodReporte, codReporte);
    removerArray(arrCodReporte, codReporte);
    spanClose.parent().remove();

    tabReportes.tabs("refresh");
    //tabReportes.tabs("option", "selected", 1);
}

function removerArray(arr) {
    var what, a = arguments, L = a.length, ax;
    while (L > 1 && arr.length) {
        what = a[--L];
        while ((ax = arr.indexOf(what)) !== -1) {
            arr.splice(ax, 1);
        }
    }
    return arr;
}
